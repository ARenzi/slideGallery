Example 2

mode: callback, direction: horizontal, step: 1, autoplay: 2sec, paging 

	var gallery2 = new slideGallery($$("div.gallery")[1], {
		steps: 1,
		autoplay: true,
		duration: 2000,
		speed: 300,
		paging: true,
		onStart: function(current, visible, length) {
			$$("span.info")[0].innerHTML = parseInt(current+1) + "-" + parseInt(visible+current) + " from " + length;
		},
		onPlay: function(current, visible, length) {
			$$("span.info")[0].innerHTML = parseInt(current+1) + "-" + parseInt(visible+current) + " from " + length;
		}
	});
